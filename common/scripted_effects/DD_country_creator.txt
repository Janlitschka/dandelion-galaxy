create_unah_republic = {
	if = {
		limit = {
			NOT = { exists = event_target:the_unah_republic }
		}
		random_galaxy_planet = {
			limit = { has_planet_flag = homeworld_unah_republic }
			create_species = {
				name = "PRESCRIPTED_species_name_unah_republic"
				class = "TOX"
				portrait = "tox6"
				homeworld = this
				namelist = "TOX1"
				traits = {
					trait = "trait_inorganic_breath"
					trait = "trait_adaptive"
					trait = "trait_weak"
					trait = "trait_slow_breeders"
					ideal_planet_class = "pc_pd_fog_desert"
				}
				plural = "PRESCRIPTED_species_plural_unah_republic"
				adjective = "PRESCRIPTED_species_adjective_unah_republic"
				gender = not_set
				effect = { save_global_event_target_as = unah_republic_Species }
			}
			last_created_species = { save_global_event_target_as = unah_republic_Species }
			create_country = {
				name = "PRESCRIPTED_name_unah_republic"
				type = default
				ignore_initial_colony_error = yes
				civics = {
					civic = "lrsk_gov_var_civic_republicanism"
					civic = "civic_merchant_guilds"
					civic = "civic_gaseous_fracking_techniques"
				}
				origin = "originsextended_heart"
				authority = "auth_democratic"
				name_list = "TOX1"
				ethos = {
					ethic = "ethic_egalitarian"
					ethic = "ethic_xenophile"
					ethic = "ethic_materialist"
				}
				species = event_target:unah_republic_Species
				flag = {
					icon = {
						category = "IBS_flag_misc_2"
						file = "z_flag_misc_2_79.dds"
					}
					background = {
						category = "backgrounds"
						file = "twins_empty_hexreverse.dds"
					}
					colors = {
						"red"
						"shadow_teal"
						"blue"
						"null"
					}
				}
				ship_prefix = "PRESCRIPTED_ship_prefix_unah_republic"
				effect = {
					set_graphical_culture = "toxoid_01"
					set_country_flag = the_unah_republic
					save_global_event_target_as = the_unah_republic
				}
			}
			set_owner = event_target:the_unah_republic
			set_capital = yes
			set_name = "PRESCRIPTED_planet_name_unah_republic"
			random_country = {
				limit = { has_country_flag = the_unah_republic }
				save_global_event_target_as = the_unah_republic
				species = { save_global_event_target_as = unah_republic_Species }
				owner_species = { set_species_homeworld = prevprev }
				country_event = { id = game_start.8 days = 0 }
				country_event = { id = game_start.9 days = 0 }
				give_starting_resources_effect = yes
				create_leader = {
					class = "official"
					species = owner_main_species
					name = "PRESCRIPTED_ruler_name_unah_republic"
					gender = female
					traits = {
						trait = "leader_trait_principled"
					}
					effect = {
						save_event_target_as = ruler_unah_republic
					}
				}
				set_leader = event_target:ruler_unah_republic
			}
			set_owner = event_target:the_unah_republic
			generate_starting_pops = { pops_species_1 = 24 }
			planet_event = { id = game_start.12 days = 0 }
			# initializer = ""
			solar_system = {
				set_name = "PRESCRIPTED_system_name_unah_republic"
				create_starbase_effect = {
					COUNTRY = unah_republic
				}
			}
		}
	}
}
